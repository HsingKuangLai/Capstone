    Set RegEx = CreateObject("VBScript.RegExp")

      .Pattern = "\d{3}-\d{1,2}"
    End With
End Sub




' 透過呼叫
Function IncrementMatches(inputString As String, Pagedif as Integer) As String
    Dim regex As Object
    Set regex = CreateObject("VBScript.RegExp")
    regex.Pattern = "\d+"
    
    Dim matches As Object
    Set matches = regex.Execute(inputString)
    
    Dim match As Object
    Dim result As String
    Dim index As Long
    Dim endIndex As Long
    
    index = 1
    For Each match In matches
        endIndex = match.FirstIndex + Len(match.Value) - 1
        result = result & Mid(inputString, index, match.FirstIndex - index + 1)
        result = result & (match.Value + Pagedif)
        index = endIndex + 2
    Next match
    
    result = result & Mid(inputString, index)
    
    IncrementMatches = result
End Function
