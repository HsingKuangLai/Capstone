Sub cleantable()

Dim RegEx As Object
    Set RegEx = CreateObject("VBScript.RegExp")
    With RegEx
      .pattern = "\d{3}-\d{1,2}"
    End With
    
Dim lr, col As Integer

lr = Cells(Rows.Count, 1).End(xlUp).Row

For col = 2 To 20

 If RegEx.test(WorksheetFunction.TextJoin("", False, Columns(col))) = True Then
 
    Dim tbrange As Range
    
    Set tbrange = Range(Cells(1, col), Cells(100, col + 10))

    tbrange.Copy

    Range("A" & CStr(lr)).PasteSpecial xlPasteAll
    
    End If
    
    Next

 Range("F1:Z" & CStr(lr)).Delete
 
On Error Resume Next
ActiveSheet.Columns("A:A").SpecialCells(xlCellTypeBlanks).EntireRow.Delete
On Error GoTo 0

Dim ws As Worksheet
    Dim lc, i As Integer
    Dim cell As Range
    Dim Regex1 As Object
    Set RegEx = CreateObject("VBScript.RegExp")
    With RegEx
      .pattern = "\d{3}-\d{1,2}"
    End With

    Set Regex1 = CreateObject("VBScript.RegExp")
    With Regex1
      .pattern = "[\u4E00-\u9FFF]"
    End With

    Set ws = ActiveSheet
    lc = ws.Cells.Find("*", SearchOrder:=xlByColumns, SearchDirection:=xlPrevious).Column
       
    For i = lc To 2 Step -1
        If Not RegEx.test(WorksheetFunction.TextJoin("", False, ws.Columns(i))) And Regex1.test(WorksheetFunction.TextJoin("", False, ws.Columns(i))) Then
            ws.Columns(i).EntireColumn.Delete Shift:=xlToLeft
            lc = lc - 1
        End If
    Next i

ActiveSheet.Columns(2).NumberFormat = "@"

End Sub
