Sub cleantable()

Dim RegEx As Object
    Set RegEx = CreateObject("VBScript.RegExp")
    With RegEx
      .Pattern = "\d{3}-\d{1,2}"
    End With
    
Dim lr As Integer

lr = Cells(rows.Count, 1).End(xlUp).Row

For col = 2 To 20

 If RegEx.test(WorksheetFunction.TextJoin("", False, Columns(col))) = True Then
 
    Dim tbrange As Range
    
    Set tbrange = Range(Cells(1, col), Cells(100, col + 10))

    tbrange.Copy

    Range("A" & CStr(lr)).PasteSpecial xlPasteAll
    
    End If
    
    Next

 Range("F1:Z" & CStr(lr)).Delete
 
On Error Resume Next
ActiveSheet.Columns("A:A").SpecialCells(xlCellTypeBlanks).EntireRow.Delete
On Error GoTo 0

Dim ws As Worksheet
    Dim lc As Long
    Dim cell As Range
    Dim Regex1 as Object
    Set RegEx = CreateObject("VBScript.RegExp")
    With RegEx
      .pattern = "\d{3}-\d{1,2}"
    End With

    Set RegEx1 = CreateObject("VBScript.RegExp")
    With RegEx1
      .pattern = "[^\u4E00-\u9FFF]"
    End With

    Set ws = ActiveSheet
    lc = ws.Cells(2, Columns.Count).End(xlToLeft).Column
       
    For Each cell In ws.Range(ws.Cells(2, 1), ws.Cells(2, lc))
        If Not RegEx.Test(cell.Value) and Not Regex.test(cell.value) Then
            cell.EntireColumn.Delete Shift:=xlToLeft
            lc = lc - 1
        End If
    Next cell

ActiveSheet.Columns(2).NumberFormat = "@"

End Sub
